1. Check the health of the Deployments with:

kubectl get deployment -lapp.kubernetes.io/instance={{ .Release.Name }}

2. Access the article-store at this URL:

if [[ ! -z "$(kubectl get pods -l app.kubernetes.io/name=external-dns -o jsonpath='{.items[*].metadata.name}')" ]]; then
  export SERVICE_HOST=$(kubectl get --namespace publisher service publisher--test--article-store -o jsonpath='{.metadata.annotations.external-dns\.alpha\.kubernetes\.io/hostname}')
else
  export SERVICE_HOST=$(kubectl get --namespace publisher service publisher--test--article-store -o jsonpath='{.status.loadBalancer.ingress[0].hostname}')
fi
echo "http://${SERVICE_HOST}"

3. Smoke test the article-store with:

helm test {{ .Release.Name }}
